# AMCL header files

set(AMCL_HEADERS
ecdh.h  
rsa.h    
utils.h) 

if(BUILD_MPIN)
  set(AMCL_HEADERS ${AMCL_HEADERS} mpin.h)
endif(BUILD_MPIN)

if(BUILD_WCC)

        OWNER_WRITE OWNER_READ OWNER_EXECUTE
        GROUP_READ GROUP_EXECUTE
        WORLD_READ WORLD_EXECUTE)

install (FILES ${AMCL_HEADERS} DESTINATION include PERMISSIONS
        OWNER_WRITE OWNER_READ OWNER_EXECUTE
        GROUP_READ GROUP_EXECUTE
        WORLD_READ WORLD_EXECUTE)
